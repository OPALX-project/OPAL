97 void SBend3D::setFieldMapFileName(std::string name) {
98  delete map_m;
99  map_m = NULL;
100 
101  if (name != "") {
102  map_m = new SectorMagneticFieldMap(
103  name,
104  "Dipole",
105  lengthUnits_m,
106  fieldUnits_m,
107  polyOrder_m,
108  smoothOrder_m);
109  double r_curv = (map_m->getPolarBoundingBoxMax()[0]+
110  map_m->getPolarBoundingBoxMin()[0])/2.;
111  double delta_phi = map_m->getDeltaPhi();
112  planarArcGeometry_m.setElementLength(r_curv*delta_phi);
113  planarArcGeometry_m.setCurvature(1./r_curv);
114  }
115 }
