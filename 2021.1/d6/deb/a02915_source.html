433 void FieldSolver::initCartesianFields() {
434 
435  e_dim_tag decomp[3] = {SERIAL, SERIAL, SERIAL};
436 
437  NDIndex<3> domain;
438  domain[0] = Index((int)getMX() + 1);
439  domain[1] = Index((int)getMY() + 1);
440  domain[2] = Index((int)getMT() + 1);
441 
442  if(Attributes::getBool(itsAttr[PARFFTX]))
443  decomp[0] = PARALLEL;
444  if(Attributes::getBool(itsAttr[PARFFTY]))
445  decomp[1] = PARALLEL;
446  if(Attributes::getBool(itsAttr[PARFFTT]))
447  decomp[2] = PARALLEL;
448 
449  if(Attributes::getString(itsAttr[FSTYPE]) == "FFTPERIODIC") {
450  decomp[0] = decomp[1] = SERIAL;
451  decomp[2] = PARALLEL;
452  }
453  // create prototype mesh and layout objects for this problem domain
454 
455  if ( !isAmrSolverType() ) {
456  mesh_m = new Mesh_t(domain);
457  FL_m = new FieldLayout_t(*mesh_m, decomp);
458  PL_m.reset(new Layout_t(*FL_m, *mesh_m));
459  // OpalData::getInstance()->setMesh(mesh_m);
460  // OpalData::getInstance()->setFieldLayout(FL_m);
461  // OpalData::getInstance()->setLayout(PL_m);
462  }

476 void FieldSolver::initSolver(PartBunchBase<double, 3> *b) {
477  itsBunch_m = b;
478  fsType_m = Attributes::getString(itsAttr[FSTYPE]);
479  std::string greens = Attributes::getString(itsAttr[GREENSF]);
480  std::string bcx = Attributes::getString(itsAttr[BCFFTX]);
481  std::string bcy = Attributes::getString(itsAttr[BCFFTY]);
482  std::string bcz = Attributes::getString(itsAttr[deprecated::BCFFTT]);
483  if (bcz == "") {
484  bcz = Attributes::getString(itsAttr[BCFFTZ]);
485  }
486 
487  if ( isAmrSolverType() ) {
488  Inform m("FieldSolver::initAmrSolver");
489  fsType_m = "AMR";
490 
491 #ifdef ENABLE_AMR
492  initAmrObject_m();
493 
494  initAmrSolver_m();
495 #endif
496  } else if(fsType_m == "FFT") {
497  bool sinTrafo = ((bcx == "DIRICHLET") && (bcy == "DIRICHLET") && (bcz == "DIRICHLET"));
498  if(sinTrafo) {
499  std::cout << "FFTBOX ACTIVE" << std::endl;
500  //we go over all geometries and add the Geometry Elements to the geometry list
501  std::string geoms = Attributes::getString(itsAttr[GEOMETRY]);
502  std::string tmp = "";
503  //split and add all to list
504  std::vector<BoundaryGeometry *> geometries;
505  for(unsigned int i = 0; i <= geoms.length(); i++) {
506  if(i == geoms.length() || geoms[i] == ',') {
507  BoundaryGeometry *geom = BoundaryGeometry::find(tmp);
508  if(geom != 0)
509  geometries.push_back(geom);
510  tmp.clear();
511  } else
512  tmp += geoms[i];
513  }
514  BoundaryGeometry *ttmp = geometries[0];
515  solver_m = new FFTBoxPoissonSolver(mesh_m, FL_m, greens, ttmp->getA());
516  itsBunch_m->set_meshEnlargement(Attributes::getReal(itsAttr[BBOXINCR]) / 100.0);
517  fsType_m = "FFTBOX";
518  } else {
519  solver_m = new FFTPoissonSolver(mesh_m, FL_m, greens, bcz);
520  itsBunch_m->set_meshEnlargement(Attributes::getReal(itsAttr[BBOXINCR]) / 100.0);
521  }
522  } else if (fsType_m == "P3M") {
523  solver_m = new P3MPoissonSolver(mesh_m,
524  FL_m,
525  Attributes::getReal(itsAttr[RC]),
526  Attributes::getReal(itsAttr[ALPHA]),
527  Attributes::getReal(itsAttr[EPSILON]));
528 
529  PL_m->setAllCacheDimensions(Attributes::getReal(itsAttr[RC]));
530  PL_m->enableCaching();
531 
532  } else if(fsType_m == "SAAMG") {
533 #ifdef HAVE_SAAMG_SOLVER
534  //we go over all geometries and add the Geometry Elements to the geometry list
535  std::string geoms = Attributes::getString(itsAttr[GEOMETRY]);
536  std::string tmp = "";
537  //split and add all to list
538  std::vector<BoundaryGeometry *> geometries;
539  for(unsigned int i = 0; i <= geoms.length(); i++) {
540  if(i == geoms.length() || geoms[i] == ',') {
541  BoundaryGeometry *geom = OpalData::getInstance()->getGlobalGeometry();
542  if(geom != 0) {
543  geometries.push_back(geom);
544  }
545  tmp.clear();
546  } else
547  tmp += geoms[i];
548  }
549  solver_m = new MGPoissonSolver(dynamic_cast<PartBunch*>(itsBunch_m), mesh_m, FL_m,
550  geometries,
551  Attributes::getString(itsAttr[ITSOLVER]),
552  Attributes::getString(itsAttr[INTERPL]),
553  Attributes::getReal(itsAttr[TOL]),
554  Attributes::getReal(itsAttr[MAXITERS]),
555  Attributes::getString(itsAttr[PRECMODE]));
556  itsBunch_m->set_meshEnlargement(Attributes::getReal(itsAttr[BBOXINCR]) / 100.0);
557 #else
558  throw OpalException("FieldSolver::initSolver",
559  "SAAMG Solver not enabled! Please build OPAL with -DENABLE_SAAMG_SOLVER=1");
560 #endif
561  } else {
562  solver_m = 0;
563  INFOMSG("no solver attached" << endl);
564  }
