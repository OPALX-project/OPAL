99 void OpalMultipoleTStraight::update() {
100  OpalElement::update();
101 
102  // Magnet length.
103  MultipoleTStraight *multT =
104  dynamic_cast<MultipoleTStraight*>(getElement());
105  double length = Attributes::getReal(itsAttr[LENGTH]);
106  double boundingBoxLength = Attributes::getReal(itsAttr[BBLENGTH]);
107  multT->setElementLength(length);
108  multT->setLength(length);
109  multT->setAperture(Attributes::getReal(itsAttr[VAPERT]),
110  Attributes::getReal(itsAttr[HAPERT]));
111 
112  multT->setFringeField(Attributes::getReal(itsAttr[LENGTH])/2,
113  Attributes::getReal(itsAttr[LFRINGE]),
114  Attributes::getReal(itsAttr[RFRINGE]));
115  multT->setBoundingBoxLength(Attributes::getReal(itsAttr[BBLENGTH]));
116  const std::vector<double> transProfile =
117  Attributes::getRealArray(itsAttr[TP]);
118  int transSize = transProfile.size();
119 
120  multT->setTransMaxOrder(transSize - 1);
121  multT->setMaxOrder(Attributes::getReal(itsAttr[MAXFORDER]));
122  multT->setRotation(Attributes::getReal(itsAttr[ROTATION]));
123  multT->setEntranceAngle(Attributes::getReal(itsAttr[EANGLE]));
124 
125  StraightGeometry &geometry = multT->getGeometry();
126 
127  geometry = StraightGeometry(2 * boundingBoxLength);
128 
129  for(int comp = 0; comp < transSize; comp++) {
130  multT->setTransProfile(comp, transProfile[comp]);
131  }
132  // Transmit "unknown" attributes.
133  OpalElement::updateUnknown(multT);
134 
135  setElement(multT);
136 }
