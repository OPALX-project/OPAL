361 template<class T, class OP>
362 struct AnyHolder
363 {
364  bool Test;
365  T Val;
366  OP Op;
367  AnyHolder() : Test(false), Val(T(0)), Op(OP()) {}
368  AnyHolder(const T& t, OP op) : Test(false), Val(t), Op(op) {}
369  AnyHolder(const AnyHolder<T,OP>& rhs)
370  : Test(rhs.Test), Val(rhs.Val), Op(rhs.Op) { }
371  const AnyHolder<T,OP>& operator=(const T& rhs)
372  {
373  if ( PETE_apply(Op,rhs,Val) )
374  Test = true;
375  return *this;
376  }
377  const AnyHolder<T,OP>& operator=(const AnyHolder<T,OP>& rhs)
378  {
379  Test = rhs.Test;
380  Val = rhs.Val;
381  Op = rhs.Op;
382  return *this;
383  }
384  const AnyHolder<T,OP>& operator*=(const T& rhs)
385  {
386  if ( PETE_apply(Op,rhs,Val) )
387  Test = true;
388  return *this;
389  }
390  const AnyHolder<T,OP>& operator*=(const AnyHolder<T,OP>& rhs)
391  {
392  Test = (Test || rhs.Test);
393  return *this;
394  }
395  Message& putMessage(Message& m)
396  {
397  m.put(Test);
398  return m;
399  }
400  Message& getMessage(Message& m)
401  {
402  m.get(Test);
403  return m;
404  }
405 };
