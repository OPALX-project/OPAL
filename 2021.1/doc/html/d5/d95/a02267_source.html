75 void OpalQuadrupole::update() {
76  OpalElement::update();
77 
78  MultipoleRep* quad =
79  dynamic_cast<MultipoleRep*>(getElement());
80 
81  quad->setElementLength(Attributes::getReal(itsAttr[LENGTH]));
82  double factor = OpalData::getInstance()->getP0() / Physics::c;
83 
84  BMultipoleField field;
85  field.setNormalComponent(2, factor * Attributes::getReal(itsAttr[K1])); // this is for the maps
86  field.setSkewComponent(2, factor * Attributes::getReal(itsAttr[K1S])); // this is for the maps
87  quad->setField(field);
88  quad->setNormalComponent(2, Attributes::getReal(itsAttr[K1]), Attributes::getReal(itsAttr[DK1]));
89  quad->setSkewComponent(2, Attributes::getReal(itsAttr[K1S]), Attributes::getReal(itsAttr[DK1S]));
90  quad->setNSlices(Attributes::getReal(itsAttr[NSLICES]));
91 
92  if (itsAttr[PARTICLEMATTERINTERACTION] && parmatint_m == NULL) {
93  const std::string matterDescriptor = Attributes::getString(itsAttr[PARTICLEMATTERINTERACTION]);
94  ParticleMatterInteraction* orig = ParticleMatterInteraction::find(matterDescriptor);
95  parmatint_m = orig->clone(matterDescriptor);
96  parmatint_m->initParticleMatterInteractionHandler(*quad);
97  quad->setParticleMatterInteraction(parmatint_m->handler_m);
98  }
99 
100  // Transmit "unknown" attributes.
101  OpalElement::updateUnknown(quad);
102 }
