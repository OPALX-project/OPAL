74 Channel *MultipoleRep::getChannel(const std::string &aKey, bool create) {
75  if(aKey[0] == 'A' || aKey[0] == 'B') {
76  int n = 0;
77 
78  for(std::string::size_type k = 1; k < aKey.length(); k++) {
79  if(isdigit(aKey[k])) {
80  n = 10 * n + aKey[k] - '0';
81  } else {
82  return 0;
83  }
84  }
85 
86  if(aKey[0] == 'B') {
87  return new IndexedChannel<MultipoleRep>
88  (*this, &MultipoleRep::getNormalComponent,
89  &MultipoleRep::setNormalComponent, n);
90  } else {
91  return new IndexedChannel<MultipoleRep>
92  (*this, &MultipoleRep::getSkewComponent,
93  &MultipoleRep::setSkewComponent, n);
94  }
95  } else {
96  for(const Entry *entry = entries; entry->name != 0; ++entry) {
97  if(aKey == entry->name) {
98  return new IndirectChannel<MultipoleRep>
99  (*this, entry->get, entry->set);
100  }
101  }
102 
103  return ElementBase::getChannel(aKey, create);
104  }
105 }
