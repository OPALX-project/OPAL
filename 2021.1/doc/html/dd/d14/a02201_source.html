105 void OpalMultipoleTCurvedConstRadius::update() {
106  OpalElement::update();
107 
108  // Magnet length.
109  MultipoleTCurvedConstRadius *multT =
110  dynamic_cast<MultipoleTCurvedConstRadius*>(getElement());
111  double length = Attributes::getReal(itsAttr[LENGTH]);
112  double angle = Attributes::getReal(itsAttr[ANGLE]);
113  double boundingBoxLength = Attributes::getReal(itsAttr[BBLENGTH]);
114  multT->setElementLength(length);
115  multT->setLength(length);
116  multT->setBendAngle(angle);
117  multT->setAperture(Attributes::getReal(itsAttr[VAPERT]),
118  Attributes::getReal(itsAttr[HAPERT]));
119 
120  multT->setFringeField(Attributes::getReal(itsAttr[LENGTH])/2,
121  Attributes::getReal(itsAttr[LFRINGE]),
122  Attributes::getReal(itsAttr[RFRINGE]));
123  multT->setBoundingBoxLength(Attributes::getReal(itsAttr[BBLENGTH]));
124  const std::vector<double> transProfile =
125  Attributes::getRealArray(itsAttr[TP]);
126  std::size_t transSize = transProfile.size();
127  if (transSize == 0) {
128  multT->setTransMaxOrder(0);
129  } else {
130  multT->setTransMaxOrder(transSize - 1);
131  }
132 
133  multT->setMaxOrder(Attributes::getReal(itsAttr[MAXFORDER]));
134  multT->setMaxXOrder(Attributes::getReal(itsAttr[MAXXORDER]));
135  multT->setRotation(Attributes::getReal(itsAttr[ROTATION]));
136  multT->setEntranceAngle(Attributes::getReal(itsAttr[EANGLE]));
137 
138  PlanarArcGeometry &geometry = multT->getGeometry();
139 
140  if(length) {
141  geometry = PlanarArcGeometry(2 * boundingBoxLength, angle / length);
142  } else {
143  geometry = PlanarArcGeometry(angle);
144  }
145 
146  for(std::size_t comp = 0; comp < transSize; comp++) {
147  multT->setTransProfile(comp, transProfile[comp]);
148  }
149  // Transmit "unknown" attributes.
150  OpalElement::updateUnknown(multT);
151 
152  setElement(multT);
153 }
