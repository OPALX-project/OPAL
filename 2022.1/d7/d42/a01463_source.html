208  + leverz * (1.0 - leverr) * FieldstrengthEr_m[index1 + 1]
209  + (1.0 - leverz) * leverr * FieldstrengthEr_m[index2]

215  + leverz * leverr * FieldstrengthEz_m[index2 + 1];
