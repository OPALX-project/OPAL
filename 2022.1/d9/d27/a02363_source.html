104 void OpalMultipoleT::update() {
105  OpalElement::update();
106 
107  // Magnet length.
108  MultipoleT *multT =
109  dynamic_cast<MultipoleT*>(getElement());
110  double length = Attributes::getReal(itsAttr[LENGTH])*Units::m2mm;
111  double angle = Attributes::getReal(itsAttr[ANGLE]);
112  multT->setElementLength(length);
113  multT->setLength(length);
114  multT->setBendAngle(angle);
115  multT->setAperture(Attributes::getReal(itsAttr[VAPERT])*Units::m2mm,
116  Attributes::getReal(itsAttr[HAPERT])*Units::m2mm);
117  multT->setFringeField(Attributes::getReal(itsAttr[LENGTH])*Units::m2mm/2,
118  Attributes::getReal(itsAttr[LFRINGE])*Units::m2mm,
119  Attributes::getReal(itsAttr[RFRINGE])*Units::m2mm);
120  if (Attributes::getBool(itsAttr[VARRADIUS])) {
121  multT->setVarRadius();
122  }
123  multT->setBoundingBoxLength(Attributes::getReal(itsAttr[BBLENGTH])*Units::m2mm);
124  const std::vector<double> transProfile =
125  Attributes::getRealArray(itsAttr[TP]);
126  int transSize = transProfile.size();
127 
128  if (transSize == 0) {
129  multT->setTransMaxOrder(0);
130  } else {
131  multT->setTransMaxOrder(transSize - 1);
132  }
133  multT->setMaxOrder(Attributes::getReal(itsAttr[MAXFORDER]));
134  multT->setMaxXOrder(Attributes::getReal(itsAttr[MAXXORDER]));
135  multT->setRotation(Attributes::getReal(itsAttr[ROTATION]));
136  multT->setEntranceAngle(Attributes::getReal(itsAttr[EANGLE]));
137 
138  for(int comp = 0; comp < transSize; comp++) {
139  double units = Units::T2kG*gsl_sf_pow_int(Units::mm2m, comp); // T m^-comp -> kG mm^-comp
140  multT->setTransProfile(comp, transProfile[comp]*units);
141  }
142  // Transmit "unknown" attributes.
143  OpalElement::updateUnknown(multT);
144  multT->initialise();
145 
146  setElement(multT);
147 }
