87 void OpalRingDefinition::update() {

91  ring->setBeamRInit(Attributes::getReal(itsAttr[BEAM_RINIT])*Units::m2mm);

93

95  ring->setLatticeThetaInit(Attributes::getReal(itsAttr[LAT_THETAINIT])*Units::deg2rad);
96  ring->setSymmetry(Attributes::getReal(itsAttr[SYMMETRY]));
97  ring->setScale(Attributes::getReal(itsAttr[SCALE]));
98 
99  ring->setHarmonicNumber(Attributes::getReal(itsAttr[HARMONIC_NUMBER]));
100  ring->setRFFreq(Attributes::getReal(itsAttr[RFFREQ]));
101  ring->setIsClosed(Attributes::getBool(itsAttr[IS_CLOSED]));
102  double minR = -1;
103  double maxR = -1;
104 
105  if (itsAttr[MIN_R]) {
106  minR = Attributes::getReal(itsAttr[MIN_R]);
107  if (!itsAttr[MAX_R]) {
108  throw (""); // EXCEPTION
109  }
110  }
111  if (itsAttr[MAX_R]) {
112  maxR = Attributes::getReal(itsAttr[MAX_R]);
113  if (!itsAttr[MIN_R]) {
114  throw (""); // EXCEPTION
115  }
116  ring->setRingAperture(minR, maxR);
117  }
118 
119  setElement(ring);
120 }
