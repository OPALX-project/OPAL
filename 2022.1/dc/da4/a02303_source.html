79 void OpalDrift::update() {
80  OpalElement::update();
81 
82  DriftRep* drf = static_cast<DriftRep*>(getElement());
83 
84  drf->setElementLength(Attributes::getReal(itsAttr[LENGTH]));
85  drf->setNSlices(Attributes::getReal(itsAttr[NSLICES]));
86 
87  if (itsAttr[WAKEF] && owk_m == nullptr) {
88  owk_m = (OpalWake::find(Attributes::getString(itsAttr[WAKEF])))->clone(getOpalName() + std::string("_wake"));
89  owk_m->initWakefunction(*drf);
90  drf->setWake(owk_m->wf_m);
91  }
92 
93  if (itsAttr[PARTICLEMATTERINTERACTION] && parmatint_m == nullptr) {
94  const std::string matterDescriptor = Attributes::getString(itsAttr[PARTICLEMATTERINTERACTION]);
95  ParticleMatterInteraction* orig = ParticleMatterInteraction::find(matterDescriptor);
96  parmatint_m = orig->clone(matterDescriptor);
97  parmatint_m->initParticleMatterInteractionHandler(*drf);
98  drf->setParticleMatterInteraction(parmatint_m->handler_m);
99  }
100 
101  if (itsAttr[GEOMETRY] && obgeo_m == nullptr) {
102  obgeo_m = (BoundaryGeometry::find(Attributes::getString(itsAttr[GEOMETRY])))->clone(getOpalName() + std::string("_geometry"));
103  if (obgeo_m) {
104  drf->setBoundaryGeometry(obgeo_m);
105  }
106  }
107 
108  // Transmit "unknown" attributes.
109  OpalElement::updateUnknown(drf);
110 }
