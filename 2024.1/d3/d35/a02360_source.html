69 void OpalSextupole::update() {
70  OpalElement::update();
71 
72  MultipoleRep *sext =
73  dynamic_cast<MultipoleRep *>(getElement());
74  sext->setElementLength(Attributes::getReal(itsAttr[LENGTH]));
75  double factor = OpalData::getInstance()->getP0() / (Physics::c * 2.0);
76  BMultipoleField field;
77  field.setNormalComponent(3, factor * Attributes::getReal(itsAttr[K2]));
78  field.setSkewComponent(3, factor * Attributes::getReal(itsAttr[K2S]));
79  sext->setField(field);
80  sext->setNormalComponent(3, Attributes::getReal(itsAttr[K2]), Attributes::getReal(itsAttr[DK2]));
81  sext->setSkewComponent(3, Attributes::getReal(itsAttr[K2S]), Attributes::getReal(itsAttr[DK2S]));
82 
83  // Transmit "unknown" attributes.
84  OpalElement::updateUnknown(sext);
85 }
