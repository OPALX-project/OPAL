97 void OpalRingDefinition::update() {
98  Ring* ring = dynamic_cast<Ring*>(getElement());
99 
100  ring->setBeamPhiInit(Attributes::getReal(itsAttr[BEAM_PHIINIT]) * Units::deg2rad);
101  ring->setBeamThetaInit(Attributes::getReal(itsAttr[BEAM_THETAINIT]) * Units::deg2rad);
102  ring->setBeamPRInit(Attributes::getReal(itsAttr[BEAM_PRINIT]));
103  ring->setBeamRInit(Attributes::getReal(itsAttr[BEAM_RINIT]));
104  ring->setLatticeRInit(Attributes::getReal(itsAttr[LAT_RINIT]));
105  ring->setLatticePhiInit(Attributes::getReal(itsAttr[LAT_PHIINIT]) * Units::deg2rad);
106  ring->setLatticeThetaInit(Attributes::getReal(itsAttr[LAT_THETAINIT]) * Units::deg2rad);
107 
108  ring->setSymmetry(Attributes::getReal(itsAttr[SYMMETRY]));
109  ring->setScale(Attributes::getReal(itsAttr[SCALE]));
110 
111  ring->setHarmonicNumber(Attributes::getReal(itsAttr[HARMONIC_NUMBER]));
112  ring->setRFFreq(Attributes::getReal(itsAttr[RFFREQ]));
113  ring->setIsClosed(Attributes::getBool(itsAttr[IS_CLOSED]));
114 
115  if (itsAttr[MIN_R] && itsAttr[MAX_R]) {
116  double minR = Attributes::getReal(itsAttr[MIN_R]);
117  double maxR = Attributes::getReal(itsAttr[MAX_R]);
118  ring->setRingAperture(minR, maxR);
119  } else if (itsAttr[MIN_R] && !itsAttr[MAX_R]) {
120  throw OpalException("OpalRingDefinition::update",
121  "If MIN_R is defined, MAX_R must also be defined.");
122  } else if (!itsAttr[MIN_R] && itsAttr[MAX_R]) {
123  throw OpalException("OpalRingDefinition::update",
124  "If MAX_R is defined, MIN_R must also be defined.");
125  }
126 
127  setElement(ring);
128 }
