339 /*. . . . . . . . . . . . . . . . . . . . . . . . . . . . . . . . . . . . . .
340 
341  Compute intersection of P1 --> P2 line segment with face planes
342  Then test intersection point to see if it is on cube face
343  Consider only face planes in "outcode_diff"
344  Note: Zero bits in "outcode_diff" means face line is outside of
345 */
346 static inline int
347 check_line (
348  const Vector_t& p1,
349  const Vector_t& p2,
350  const int outcode_diff
351  ) {
352  if ((0x01 & outcode_diff) != 0)
353  if (check_point(p1,p2,( .5-p1[0])/(p2[0]-p1[0]),0x3e) == INSIDE) return(INSIDE);
354  if ((0x02 & outcode_diff) != 0)

356  if ((0x04 & outcode_diff) != 0)
357  if (check_point(p1,p2,( .5-p1[1])/(p2[1]-p1[1]),0x3b) == INSIDE) return(INSIDE);
358  if ((0x08 & outcode_diff) != 0)
359  if (check_point(p1,p2,(-.5-p1[1])/(p2[1]-p1[1]),0x37) == INSIDE) return(INSIDE);
360  if ((0x10 & outcode_diff) != 0)
361  if (check_point(p1,p2,( .5-p1[2])/(p2[2]-p1[2]),0x2f) == INSIDE) return(INSIDE);
362  if ((0x20 & outcode_diff) != 0)
363  if (check_point(p1,p2,(-.5-p1[2])/(p2[2]-p1[2]),0x1f) == INSIDE) return(INSIDE);

381 static int
382 point_triangle_intersection (
383  const Vector_t& p,
384  const Triangle& t
385  ) {
386  /*
387  First, a quick bounding-box test:
388  If P is outside triangle bbox, there cannot be an intersection.
389  */
390  if (cmp::gt(p[0], std::max({t.v1(0), t.v2(0), t.v3(0)}))) return(OUTSIDE);
391  if (cmp::gt(p[1], std::max({t.v1(1), t.v2(1), t.v3(1)}))) return(OUTSIDE);
392  if (cmp::gt(p[2], std::max({t.v1(2), t.v2(2), t.v3(2)}))) return(OUTSIDE);
393  if (cmp::lt(p[0], std::min({t.v1(0), t.v2(0), t.v3(0)}))) return(OUTSIDE);
394  if (cmp::lt(p[1], std::min({t.v1(1), t.v2(1), t.v3(1)}))) return(OUTSIDE);
395  if (cmp::lt(p[2], std::min({t.v1(2), t.v2(2), t.v3(2)}))) return(OUTSIDE);
396 
397  /*
398  For each triangle side, make a vector out of it by subtracting vertexes;
399  make another vector from one vertex to point P.
400  The crossproduct of these two vectors is orthogonal to both and the
401  signs of its X,Y,Z components indicate whether P was to the inside or
402  to the outside of this triangle side.
403  */
404  const Vector_t vect12 = t.v1() - t.v2();
405  const Vector_t vect1h = t.v1() - p;
406  const Vector_t cross12_1p = cross (vect12, vect1h);
407  const int sign12 = SIGN3(cross12_1p); /* Extract X,Y,Z signs as 0..7 or 0...63 integer */
408 
409  const Vector_t vect23 = t.v2() - t.v3();
410  const Vector_t vect2h = t.v2() - p;
411  const Vector_t cross23_2p = cross (vect23, vect2h);
412  const int sign23 = SIGN3(cross23_2p);
413 
414  const Vector_t vect31 = t.v3() - t.v1();
415  const Vector_t vect3h = t.v3() - p;
416  const Vector_t cross31_3p = cross (vect31, vect3h);
417  const int sign31 = SIGN3(cross31_3p);
418 
419  /*
420  If all three crossproduct vectors agree in their component signs,
421  then the point must be inside all three.
422  P cannot be OUTSIDE all three sides simultaneously.
423  */
424  return ((sign12 & sign23 & sign31) == 0) ? OUTSIDE : INSIDE;
425 }

428 /*. . . . . . . . . . . . . . . . . . . . . . . . . . . . . . . . . . . . . .
429 
430  This is the main algorithm procedure.
431  Triangle t is compared with a unit cube,
432  centered on the origin.
433  It returns INSIDE (0) or OUTSIDE(1) if t
434  intersects or does not intersect the cube.
435 */
436 static int
437 triangle_intersects_cube (
438  const Triangle& t
439  ) {
440  int v1_test;
441  int v2_test;
442  int v3_test;
443 
444  /*
445  First compare all three vertexes with all six face-planes
446  If any vertex is inside the cube, return immediately!
447  */
448  if ((v1_test = face_plane(t.v1())) == INSIDE) return(INSIDE);
449  if ((v2_test = face_plane(t.v2())) == INSIDE) return(INSIDE);
450  if ((v3_test = face_plane(t.v3())) == INSIDE) return(INSIDE);
451 
452  /*
453  If all three vertexes were outside of one or more face-planes,
454  return immediately with a trivial rejection!
455  */
456  if ((v1_test & v2_test & v3_test) != 0) return(OUTSIDE);
457 
458  /*
459  Now do the same trivial rejection test for the 12 edge planes
460  */
461  v1_test |= bevel_2d(t.v1()) << 8;
462  v2_test |= bevel_2d(t.v2()) << 8;
463  v3_test |= bevel_2d(t.v3()) << 8;
464  if ((v1_test & v2_test & v3_test) != 0) return(OUTSIDE);
465 
466  /*
467  Now do the same trivial rejection test for the 8 corner planes
468  */
469  v1_test |= bevel_3d(t.v1()) << 24;
470  v2_test |= bevel_3d(t.v2()) << 24;
471  v3_test |= bevel_3d(t.v3()) << 24;
472  if ((v1_test & v2_test & v3_test) != 0) return(OUTSIDE);
473 
474  /*
475  If vertex 1 and 2, as a pair, cannot be trivially rejected
476  by the above tests, then see if the v1-->v2 triangle edge
477  intersects the cube. Do the same for v1-->v3 and v2-->v3./
478  Pass to the intersection algorithm the "OR" of the outcode
479  bits, so that only those cube faces which are spanned by
480  each triangle edge need be tested.
481  */
482  if ((v1_test & v2_test) == 0)
483  if (check_line (t.v1(), t.v2(), v1_test|v2_test) == INSIDE) return(INSIDE);
484  if ((v1_test & v3_test) == 0)
485  if (check_line (t.v1(), t.v3(), v1_test|v3_test) == INSIDE) return(INSIDE);
486  if ((v2_test & v3_test) == 0)
487  if (check_line (t.v2(), t.v3(), v2_test|v3_test) == INSIDE) return(INSIDE);
488 
489  /*
490  By now, we know that the triangle is not off to any side,
491  and that its sides do not penetrate the cube. We must now
492  test for the cube intersecting the interior of the triangle.
493  We do this by looking for intersections between the cube
494  diagonals and the triangle...first finding the intersection
495  of the four diagonals with the plane of the triangle, and
496  then if that intersection is inside the cube, pursuing
497  whether the intersection point is inside the triangle itself.
498 
499  To find plane of the triangle, first perform crossproduct on
500  two triangle side vectors to compute the normal vector.
501  */
502  Vector_t vect12 = t.v1() - t.v2();
503  Vector_t vect13 = t.v1() - t.v3();
504  Vector_t norm = cross (vect12, vect13);
505 
506  /*
507  The normal vector "norm" X,Y,Z components are the coefficients
508  of the triangles AX + BY + CZ + D = 0 plane equation. If we
509  solve the plane equation for X=Y=Z (a diagonal), we get
510  -D/(A+B+C) as a metric of the distance from cube center to the
511  diagonal/plane intersection. If this is between -0.5 and 0.5,
512  the intersection is inside the cube. If so, we continue by
513  doing a point/triangle intersection.
514  Do this for all four diagonals.
515  */
516  double d = norm[0] * t.v1(0) + norm[1] * t.v1(1) + norm[2] * t.v1(2);
517 
518  /*
519  if one of the diagonals is parallel to the plane, the other will
520  intersect the plane
521  */
522  double denom = norm[0] + norm[1] + norm[2];
523  if (cmp::eq_zero(std::abs(denom)) == false) {
524  /* skip parallel diagonals to the plane; division by 0 can occure */
525  Vector_t hitpp = d / denom;
526  if (cmp::le(std::abs(hitpp[0]), 0.5))
527  if (point_triangle_intersection(hitpp,t) == INSIDE)
528  return(INSIDE);
529  }
530  denom = norm[0] + norm[1] - norm[2];
531  if (cmp::eq_zero(std::abs(denom)) == false) {
532  Vector_t hitpn;
533  hitpn[2] = -(hitpn[0] = hitpn[1] = d / denom);
534  if (cmp::le(std::abs(hitpn[0]), 0.5))
535  if (point_triangle_intersection(hitpn,t) == INSIDE)
536  return(INSIDE);
537  }
538  denom = norm[0] - norm[1] + norm[2];
539  if (cmp::eq_zero(std::abs(denom)) == false) {
540  Vector_t hitnp;
541  hitnp[1] = -(hitnp[0] = hitnp[2] = d / denom);
542  if (cmp::le(std::abs(hitnp[0]), 0.5))
543  if (point_triangle_intersection(hitnp,t) == INSIDE)
544  return(INSIDE);
545  }
546  denom = norm[0] - norm[1] - norm[2];
547  if (cmp::eq_zero(std::abs(denom)) == false) {
548  Vector_t hitnn;
549  hitnn[1] = hitnn[2] = -(hitnn[0] = d / denom);
550  if (cmp::le(std::abs(hitnn[0]), 0.5))
551  if (point_triangle_intersection(hitnn,t) == INSIDE)
552  return(INSIDE);
553  }
554 
555  /*
556  No edge touched the cube; no cube diagonal touched the triangle.
557  We're done...there was no intersection.
558  */
559  return(OUTSIDE);
560 }

951 BoundaryGeometry::intersectLineTriangle (
952  const enum INTERSECTION_TESTS kind,
953  const Vector_t& P0,
954  const Vector_t& P1,
955  const int triangle_id,
956  Vector_t& I
957  ) {
958  const Vector_t V0 = getPoint (triangle_id, 1);
959  const Vector_t V1 = getPoint (triangle_id, 2);
960  const Vector_t V2 = getPoint (triangle_id, 3);
961 
962  // get triangle edge vectors and plane normal
963  const Vector_t u = V1 - V0; // triangle vectors
964  const Vector_t v = V2 - V0;
965  const Vector_t n = cross (u, v);
966  if (n == (Vector_t)0) // triangle is degenerate
967  return -1; // do not deal with this case
968 
969  const Vector_t dir = P1 - P0; // ray direction vector
970  const Vector_t w0 = P0 - V0;
971  const double a = -dot(n,w0);
972  const double b = dot(n,dir);
973  if (cmp::eq_zero(b)) { // ray is parallel to triangle plane
974  if (cmp::eq_zero(a)) { // ray lies in triangle plane
975  return 1;
976  } else { // ray disjoint from plane
977  return 0;
978  }
979  }
980 
981  // get intersect point of ray with triangle plane
982  const double r = a / b;
983  switch (kind) {
984  case RAY:
985  if (cmp::lt_zero(r)) { // ray goes away from triangle
986  return 0; // => no intersect
987  }
988  break;
989  case SEGMENT:
990  if (cmp::lt_zero(r) || cmp::lt(1.0, r)) { // intersection on extended
991  return 0; // segment
992  }
993  break;
994  case LINE:
995  break;
996  };
997  I = P0 + r * dir; // intersect point of ray and plane
998 
999  // is I inside T?
1000  const double uu = dot(u,u);
1001  const double uv = dot(u,v);
1002  const double vv = dot(v,v);
1003  const Vector_t w = I - V0;
1004  const double wu = dot(w,u);
1005  const double wv = dot(w,v);
1006  const double D = uv * uv - uu * vv;
1007 
1008  // get and test parametric coords
1009  const double s = (uv * wv - vv * wu) / D;
1010  if (cmp::lt_zero(s) || cmp::gt(s, 1.0)) { // I is outside T
1011  return 0;
1012  }
1013  const double t = (uv * wu - uu * wv) / D;
1014  if (cmp::lt_zero(t) || cmp::gt((s + t), 1.0)) { // I is outside T
1015  return 0;
1016  }
1017  // intersection point is in triangle
1018  if (cmp::lt_zero(r)) { // in extended segment in opposite
1019  return 2; // direction of ray
1020  } else if (cmp::ge_zero(r) && cmp::le(r, 1.0)) { // in segment
1021  return 3;
1022  } else { // in extended segment in
1023  return 4; // direction of ray
1024  }

1034 BoundaryGeometry::isInside (
1035  const Vector_t& P // [in] pt to test
1036  ) {

1038  /*
1039  select a "close" reference pt outside the bounding box
1040  */
1041  // right boundary of bounding box (x direction)
1042  double x = minExtent_m[0] - 0.01;
1043  double distance = P[0] - x;
1044  Vector_t ref_pt {x, P[1], P[2]};
1045 
1046  // left boundary of bounding box (x direction)
1047  x = maxExtent_m[0] + 0.01;
1048  if (cmp::lt(x - P[0], distance)) {
1049  distance = x - P[0];
1050  ref_pt = {x, P[1], P[2]};
1051  }
1052 
1053  // lower boundary of bounding box (y direction)
1054  double y = minExtent_m[1] - 0.01;
1055  if (cmp::lt(P[1] - y, distance)) {
1056  distance = P[1] -y;
1057  ref_pt = {P[0], y, P[1]};
1058  }
1059 
1060  // upper boundary of bounding box (y direction)
1061  y = maxExtent_m[1] + 0.01;
1062  if (cmp::lt(y - P[1], distance)) {
1063  distance = y - P[1];
1064  ref_pt = {P[0], y, P[2]};
1065  }
1066  // front boundary of bounding box (z direction)
1067  double z = minExtent_m[2] - 0.01;
1068  if (cmp::lt(P[2] - z, distance)) {
1069  distance = P[2] - z;
1070  ref_pt = {P[0], P[1], z};
1071  }
1072  // back boundary of bounding box (z direction)
1073  z = maxExtent_m[2] + 0.01;
1074  if (cmp::lt(z - P[2], distance)) {
1075  ref_pt = {P[0], P[1], z};
1076  }
1077 
1078  /*
1079  the test returns the number of intersections =>
1080  since the reference point is outside, P is inside
1081  if the result is odd.
1082  */
1083  int k = fastIsInside (ref_pt, P);
1084  return (k % 2) == 1;
1085 }
1086 
1087 /*
1088  searching a point inside the geometry.
1089 
1090  sketch of the algorithm:
1091  In a first step, we try to find a line segment defined by one
1092  point outside the bounding box and a point somewhere inside the
1093  bounding box which has intersects with the geometry.
1094 
1095  If the number of intersections is odd, the center point is inside
1096  the geometry and we are already done.
1097 
1098  If the number of intersections is even, there must be points on
1099  this line segment which are inside the geometry. In the next step
1100  we have to find one if these points.
1101 
1102 
1103  A bit more in detail:
1104 
1105  1. Finding a line segment intersecting the geometry
1106  For the fast isInside test it is of advantage to choose line segments
1107  parallel to the X, Y or Z axis. In this implementation we choose as
1108  point outside the bounding box a point on an axis but close to the
1109  bounding box and the center of the bounding box. This gives us six
1110  line segments to test. This covers not all possible geometries but
1111  most likely almost all. If not, it's easy to extend.
1112 
1113  2. Searching for a point inside the geometry
1114  In the first step we get a line segment from which we know, that one
1115  point is ouside the geometry (P_out) and the other inside the bounding
1116  box (Q). We also know the number of intersections n_i of this line
1117  segment with the geometry.
1118 
1119  If n_i is odd, Q is inside the boundary!
1120 
1121  while (true); do
1122  bisect the line segment [P_out, Q], let B the bisecting point.
1123 
1124  compute number of intersections of the line segment [P_out, B]
1125  and the geometry.
1126 
1127  If the number of intersections is odd, then B is inside the geometry
1128  and we are done. Set P_in = B and exit loop.
1129 
1130  Otherwise we have either no or an even number of intersections.
1131  In both cases this implies that B is a point outside the geometry.
1132 
1133  If the number of intersection of [P_out, B] is even but not equal zero,
1134  it might be that *all* intersections are in this line segment and none in
1135  [B, Q].
1136  In this case we continue with the line segment [P_out, Q] = [P_out, B],
1137  otherwise with the line segment [P_out, Q] = [B, Q].
1138 */
1139 bool
1140 BoundaryGeometry::findInsidePoint (
1141  void
1142  ) {
1143  *gmsg << level2 << "* Searching for a point inside the geometry..." << endl;
1144  /*
1145  find line segment
1146  */
1147  Vector_t Q {(maxExtent_m + minExtent_m) / 2};
1148  std::vector<Vector_t> P_outs {
1149  {minExtent_m[0]-0.01, Q[1], Q[2]},
1150  {maxExtent_m[0]+0.01, Q[1], Q[2]},
1151  {Q[0], minExtent_m[1]-0.01, Q[2]},
1152  {Q[0], maxExtent_m[1]+0.01, Q[2]},
1153  {Q[0], Q[1], minExtent_m[2]-0.01},
1154  {Q[0], Q[1], maxExtent_m[2]+0.01}
1155  };
1156  int n_i = 0;
1157  Vector_t P_out;
1158  for (const auto& P: P_outs) {
1159  n_i = fastIsInside (P, Q);
1160  if (n_i != 0) {
1161  P_out = P;
1162  break;
1163  }
1164  }

1974  intersection test more than once. We must this into account when
1975  counting the intersections with the boundary.

1978  - build a set of all relevant triangles and loop over this set

2006  the to be tested triangles.
2007  */

2010  voxelMesh_m.ids.find (voxel_id);

2021  if there is more than one intersection, return closest
2022  */

2024  int tmp_intersect_result = 0;
2025

2027  it != triangle_ids.end ();
2028  it++) {

2031  LINE,

2055  t = (tmp_intersect_pt[0] - P[0]) / (Q[0] - P[0]);
2056  } else if (cmp::eq_zero(Q[1] - P[1]) == false) {
2057  t = (tmp_intersect_pt[1] - P[1]) / (Q[1] - P[1]);
2058  } else {
2059  t = (tmp_intersect_pt[2] - P[2]) / (Q[2] - P[2]);
2060  }

2063 #ifdef ENABLE_DEBUG

2067  << endl;
