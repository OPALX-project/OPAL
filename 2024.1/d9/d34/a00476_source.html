164  TEMP_ITER b_end) {
165  TEMP_ITER a_it = a_begin;
166  TEMP_ITER b_it = b_begin;
167  while (a_it != a_end && b_it != b_end) {
168  if (*a_it != *b_it) return false;
169  ++a_it;
170  ++b_it;
171  }
172  if ( a_it != a_end || b_it != b_end ) return false;
173  return true;
174 }
