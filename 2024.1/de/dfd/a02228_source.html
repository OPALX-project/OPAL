99 void OpalMultipoleT::update() {
100  OpalElement::update();
101 
102  MultipoleT* multT = dynamic_cast<MultipoleT*>(getElement());
103 
104  double length = Attributes::getReal(itsAttr[LENGTH]);
105  double angle = Attributes::getReal(itsAttr[ANGLE]);
106  multT->setElementLength(length);
107  multT->setLength(length);
108  multT->setBendAngle(angle);
109  multT->setAperture(Attributes::getReal(itsAttr[VAPERT]),
110  Attributes::getReal(itsAttr[HAPERT]));
111  multT->setFringeField(Attributes::getReal(itsAttr[LENGTH]) * 0.5,
112  Attributes::getReal(itsAttr[LFRINGE]),
113  Attributes::getReal(itsAttr[RFRINGE]));
114  if (Attributes::getBool(itsAttr[VARRADIUS])) {
115  multT->setVarRadius();
116  }
117  multT->setBoundingBoxLength(Attributes::getReal(itsAttr[BBLENGTH]));
118  const std::vector<double> transProfile = Attributes::getRealArray(itsAttr[TP]);
119  int transSize = transProfile.size();
120 
121  if (transSize == 0) {
122  multT->setTransMaxOrder(0);
123  } else {
124  multT->setTransMaxOrder(transSize - 1);
125  }
126  multT->setMaxOrder(Attributes::getReal(itsAttr[MAXFORDER]));
127  multT->setMaxXOrder(Attributes::getReal(itsAttr[MAXXORDER]));
128  multT->setRotation(Attributes::getReal(itsAttr[ROTATION]));
129  multT->setEntranceAngle(Attributes::getReal(itsAttr[EANGLE]));
130 
131  for (int comp = 0; comp < transSize; comp++) {
132  double units = Units::T2kG * gsl_sf_pow_int(1.0, comp); // T m^-comp -> kG mm^-comp
133  multT->setTransProfile(comp, transProfile[comp] * units);
134  }
135  // Transmit "unknown" attributes.
136  OpalElement::updateUnknown(multT);
137  multT->initialise();
138 
139  setElement(multT);
140 }
