77 void OpalSplineTimeDependence::update() {
78  const double order = Attributes::getReal(itsAttr[ORDER]);
79  if (order != 1.0 && order != 3.0) {
80  throw OpalException("OpalSplineTimeDependence::update",
81  "SPLINE_TIME_DEPENDENCE \"ORDER\" should be 1 or 3.");
82  }
83  // Note we set array defaults as it seems that the default object must be valid
84  std::vector<double> times;
85  if (itsAttr[TIMES]) {
86  times = Attributes::getRealArray(itsAttr[TIMES]);
87  } else {
88  times = {0.0, 1.0}; // A default time array
89  }
90  std::vector<double> values;
91  if (itsAttr[VALUES]) {
92  values = Attributes::getRealArray(itsAttr[VALUES]);
93  } else {
94  values = {0.0, 0.0}; // A default value array
95  }
96  const auto spline = std::make_shared<SplineTimeDependence>(static_cast<size_t>(order),
97  times,values);
98  AbstractTimeDependence::setTimeDependence(getOpalName(), spline);
99 }
