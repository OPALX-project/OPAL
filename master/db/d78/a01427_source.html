334 {
335  if( lhs.num_row() != rhs.num_row() || lhs.num_col() != rhs.num_col() ) return false;

337  for(size_t j=1; j<=lhs.num_col(); j++)
338  if(lhs(i,j) != rhs(i,j)) return false;
