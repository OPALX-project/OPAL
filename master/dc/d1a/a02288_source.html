74 void OpalPolynomialTimeDependence::update() {
75  // getOpalName() comes from AbstractObjects/Object.h
76  if (itsAttr[COEFFICIENTS] && (itsAttr[P0] || itsAttr[P1] || itsAttr[P2] || itsAttr[P3])) {
77  // Only use P0..P3 or COEFFICIENTS, not both
78  throw OpalException("OpalPolynomialTimeDependence::Update",
79  "Use P0..P3 or COEFFICIENTS to specify the coefficients, not both.");
80  }
81  std::vector<double> polynomial_coefficients = Attributes::getRealArray(itsAttr[COEFFICIENTS]);
82  if (polynomial_coefficients.empty()) {
83  polynomial_coefficients.push_back(Attributes::getReal(itsAttr[P0]));
84  polynomial_coefficients.push_back(Attributes::getReal(itsAttr[P1]));
85  polynomial_coefficients.push_back(Attributes::getReal(itsAttr[P2]));
86  polynomial_coefficients.push_back(Attributes::getReal(itsAttr[P3]));
87  }
88  AbstractTimeDependence::setTimeDependence(
89  getOpalName(), std::make_shared<PolynomialTimeDependence>(polynomial_coefficients));
90 }
