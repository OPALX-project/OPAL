73 void OpalUndulator::update() {
74  OpalElement::update();
75 
76  UndulatorRep* ur = static_cast<UndulatorRep*>(getElement());
77  /* The element length is given by the number and length of the undulator periods, plus the
78  * length of the two fringe fields, each measuring 2 * lambda. */
79  ur->setElementLength(
80  Attributes::getReal(itsAttr[LAMBDA]) * (4 + Attributes::getReal(itsAttr[NUMPERIODS])));
81  ur->setK(Attributes::getReal(itsAttr[K]));
82  ur->setLambda(Attributes::getReal(itsAttr[LAMBDA]));
83  ur->setNumPeriods(Attributes::getReal(itsAttr[NUMPERIODS]));
84  ur->setAngle(Attributes::getReal(itsAttr[ANGLE]));
85  ur->setFilename(Attributes::getString(itsAttr[FNAME]));
86  ur->setMeshLength(Attributes::getRealArray(itsAttr[MESHLENGTH]));
87  ur->setMeshResolution(Attributes::getRealArray(itsAttr[MESHRESOLUTION]));
88  ur->setTruncationOrder(Attributes::getReal(itsAttr[TRUNORDER]));
89  ur->setTotalTime(Attributes::getReal(itsAttr[TOTALTIME]));
90  ur->setDtBunch(Attributes::getReal(itsAttr[DTBUNCH]));
91 
92  // Transmit "unknown" attributes.
93  OpalElement::updateUnknown(ur);
